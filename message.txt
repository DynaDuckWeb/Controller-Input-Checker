// ─────────────────────────────────────────────────────────────────────────────
// ANSI helpers
// ─────────────────────────────────────────────────────────────────────────────
#define ESC "\033["
#define RESET "\033[0m"
#define BOLD "\033[1m"
#define DIM "\033[2m"

// 256-color fg / bg
#define FG(n) "\033[38;5;" #n "m"
#define BG(n) "\033[48;5;" #n "m"

// Palette
#define C_BG "\033[48;5;234m"      // near-black bg
#define C_PANEL "\033[48;5;236m"   // card bg
#define C_BORDER "\033[38;5;240m"  // dim border
#define C_TITLE "\033[38;5;75m"    // blue title
#define C_LABEL "\033[38;5;250m"   // light grey label
#define C_DIM "\033[38;5;240m"     // dark grey
#define C_ON "\033[38;5;82m"       // bright green  (pressed)
#define C_AXIS "\033[38;5;75m"     // blue axis value
#define C_TRIG "\033[38;5;214m"    // orange trigger
#define C_DPAD "\033[38;5;183m"    // lavender dpad
#define C_WARN "\033[38;5;196m"    // red

static void clear_screen() { std::fputs("\033[2J\033[H", stdout); }
static void cursor_home() { std::fputs("\033[H", stdout); }
static void hide_cursor() { std::fputs("\033[?25l", stdout); }
static void show_cursor() { std::fputs("\033[?25h", stdout); }
static void goto_xy(int r, int c) { std::printf("\033[%d;%dH", r, c); }

// ─────────────────────────────────────────────────────────────────────────────
// State
// ─────────────────────────────────────────────────────────────────────────────
static constexpr int DEADZONE = 8000;

struct ControllerState {
    // Sticks  -32767…32767
    int lx = 0, ly = 0;
    int rx = 0, ry = 0;
    // Triggers  0…255
    int lt = 0, rt = 0;
    // D-Pad  -1,0,1
    int dpx = 0, dpy = 0;
    // Buttons
    std::map<WORD, bool> buttons;
    // Device name
    std::string name;
    // Last event log (short)
    std::vector<std::string> log;

    void push_log(const std::string& s) {
        log.push_back(s);
        if ((int)log.size() > 5)
            log.erase(log.begin());
    }
};

static ControllerState g_state;
static volatile bool g_running = true;

BOOL WINAPI console_ctrl_handler(DWORD ctrl_type) {
    if (ctrl_type == CTRL_C_EVENT || ctrl_type == CTRL_CLOSE_EVENT) {
        g_running = false;
        return TRUE;
    }
    return FALSE;
}

// ─────────────────────────────────────────────────────────────────────────────
// Button name table (XInput mappings)
// ─────────────────────────────────────────────────────────────────────────────
static const std::map<WORD, std::string> BTN_NAMES = {
    {XINPUT_GAMEPAD_A, "A"},
    {XINPUT_GAMEPAD_B, "B"},
    {XINPUT_GAMEPAD_X, "X"},
    {XINPUT_GAMEPAD_Y, "Y"},
    {XINPUT_GAMEPAD_LEFT_SHOULDER, "L1"},
    {XINPUT_GAMEPAD_RIGHT_SHOULDER, "R1"},
    {XINPUT_GAMEPAD_BACK, "Select"},
    {XINPUT_GAMEPAD_START, "Start"},
    {XINPUT_GAMEPAD_LEFT_THUMB, "L3"},
    {XINPUT_GAMEPAD_RIGHT_THUMB, "R3"},
    {XINPUT_GAMEPAD_DPAD_UP, "DU"},
    {XINPUT_GAMEPAD_DPAD_DOWN, "DD"},
    {XINPUT_GAMEPAD_DPAD_LEFT, "DL"},
    {XINPUT_GAMEPAD_DPAD_RIGHT, "DR"},
};
static std::string btn_name(WORD c) {
    auto it = BTN_NAMES.find(c);
    return (it != BTN_NAMES.end()) ? it->second : "?";
}

// ─────────────────────────────────────────────────────────────────────────────
// Draw helpers
// ─────────────────────────────────────────────────────────────────────────────

// Draw a horizontal bar for a trigger (0-255), width chars wide
static void draw_trigger_bar(int val, int width) {
    int filled = val * width / 255;
    std::fputs(C_TRIG, stdout);
    for (int i = 0; i < width; i++)
        std::fputs(i < filled ? "█" : "░", stdout);
    std::fputs(RESET, stdout);
}

// Draw a mini stick circle (5×3 chars) with a dot showing position
//   x,y in -32767…32767
static void draw_stick(int x, int y, int base_row, int base_col) {
    // Map -32767..32767 → 0..4 (col) and 0..2 (row)
    int cx = (x + 32767) * 4 / 65534;  // 0-4
    int cy = (y + 32767) * 2 / 65534;  // 0-2

    // Dead zone → center
    if (std::abs(x) < XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE &&
        std::abs(y) < XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE) {
        cx = 2;
        cy = 1;
    }

    const char* ring[3][5] = {
        {"╭", "─", "─", "─", "╮"},
        {"│", " ", " ", " ", "│"},
        {"╰", "─", "─", "─", "╯"},
    };
    for (int row = 0; row < 3; row++) {
        goto_xy(base_row + row, base_col);
        std::fputs(C_BORDER, stdout);
        for (int col = 0; col < 5; col++) {
            if (row == cy && col == cx)
                std::printf(C_AXIS "●" C_BORDER);
            else
                std::fputs(ring[row][col], stdout);
        }
        std::fputs(RESET, stdout);
    }
}

// Draw a D-Pad (5×5 chars) at base_row, base_col
static void draw_dpad(int dx, int dy, int base_row, int base_col) {
    auto color = [](bool on) { return on ? C_DPAD BOLD : C_DIM; };

    goto_xy(base_row, base_col);
    std::printf("  %s▲" RESET "  ", color(dy > 0));
    goto_xy(base_row + 1, base_col);
    std::printf("%s◄" RESET "   %s►" RESET, color(dx < 0), color(dx > 0));
    goto_xy(base_row + 2, base_col);
    std::printf("  %s▼" RESET "  ", color(dy < 0));
}

// Draw a button badge (label, is_pressed)
static void draw_btn(int row, int col, const std::string& label, bool pressed) {
    goto_xy(row, col);
    if (pressed)
        std::printf(C_ON BOLD "[%s]" RESET, label.c_str());
    else
        std::printf(C_DIM "[%s]" RESET, label.c_str());
}

// ─────────────────────────────────────────────────────────────────────────────
// Full dashboard render  (fits in 24×80)
// ─────────────────────────────────────────────────────────────────────────────
static void render() {
    cursor_home();

    // ── Title bar ────────────────────────────────────────────────────────────
    std::printf(
        C_PANEL BOLD C_TITLE " ◈  Controller Input  " RESET C_PANEL C_DIM
        "  %-50s" RESET "\n",
        g_state.name.c_str());
    std::printf(
        C_DIM
        "─────────────────────────────────────────────────────────────────────────────\n" RESET);

    // ── Left Trigger  ────────────────────────────────────────────────────────
    goto_xy(3, 2);
    std::printf(C_LABEL " L2 " RESET);
    goto_xy(3, 6);
    draw_trigger_bar(g_state.lt, 12);
    std::printf(C_DIM " %3d" RESET, g_state.lt * 100 / 255);
    std::fputs("%", stdout);

    // ── Right Trigger ────────────────────────────────────────────────────────
    goto_xy(3, 46);
    std::printf(C_LABEL "R2 " RESET);
    goto_xy(3, 49);
    draw_trigger_bar(g_state.rt, 12);
    std::printf(C_DIM " %3d" RESET, g_state.rt * 100 / 255);
    std::fputs("%", stdout);

    // ── Shoulder buttons ─────────────────────────────────────────────────────
    draw_btn(4, 2, "L1", g_state.buttons[XINPUT_GAMEPAD_LEFT_SHOULDER]);
    draw_btn(4, 46, "R1", g_state.buttons[XINPUT_GAMEPAD_RIGHT_SHOULDER]);

    // ── Left stick ───────────────────────────────────────────────────────────
    goto_xy(6, 2);
    std::printf(C_LABEL "Left Stick" RESET);
    draw_stick(g_state.lx, g_state.ly, 7, 2);

    std::string ldir = "";
    bool lin_dead =
        std::abs(g_state.lx) < XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE &&
        std::abs(g_state.ly) < XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE;
    if (!lin_dead) {
        if (g_state.ly > XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE)
            ldir += "UP ";
        if (g_state.ly < -XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE)
            ldir += "DOWN ";
        if (g_state.lx < -XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE)
            ldir += "LEFT";
        if (g_state.lx > XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE)
            ldir += "RIGHT";
    }
    goto_xy(10, 2);
    std::printf(C_AXIS "%-10s" RESET, lin_dead ? "CENTER" : ldir.c_str());

    // ── Right stick ──────────────────────────────────────────────────────────
    goto_xy(6, 46);
    std::printf(C_LABEL "Right Stick" RESET);
    draw_stick(g_state.rx, g_state.ry, 7, 46);

    std::string rdir = "";
    bool rin_dead =
        std::abs(g_state.rx) < XINPUT_GAMEPAD_RIGHT_THUMB_DEADZONE &&
        std::abs(g_state.ry) < XINPUT_GAMEPAD_RIGHT_THUMB_DEADZONE;
    if (!rin_dead) {
        if (g_state.ry > XINPUT_GAMEPAD_RIGHT_THUMB_DEADZONE)
            rdir += "UP ";
        if (g_state.ry < -XINPUT_GAMEPAD_RIGHT_THUMB_DEADZONE)
            rdir += "DOWN ";
        if (g_state.rx < -XINPUT_GAMEPAD_RIGHT_THUMB_DEADZONE)
            rdir += "LEFT";
        if (g_state.rx > XINPUT_GAMEPAD_RIGHT_THUMB_DEADZONE)
            rdir += "RIGHT";
    }
    goto_xy(10, 46);
    std::printf(C_AXIS "%-10s" RESET, rin_dead ? "CENTER" : rdir.c_str());

    // ── D-Pad ────────────────────────────────────────────────────────────────
    goto_xy(6, 22);
    std::printf(C_LABEL "D-Pad" RESET);
    draw_dpad(g_state.dpx, g_state.dpy, 7, 22);

    // ── Face buttons ─────────────────────────────────────────────────────────
    goto_xy(7, 34);
    std::printf("  ");
    draw_btn(7, 36, "Y", g_state.buttons[XINPUT_GAMEPAD_Y]);
    draw_btn(8, 34, "X", g_state.buttons[XINPUT_GAMEPAD_X]);
    goto_xy(8, 38);
    std::fputs("  ", stdout);
    draw_btn(8, 38, "B", g_state.buttons[XINPUT_GAMEPAD_B]);
    goto_xy(9, 35);
    std::fputs(" ", stdout);
    draw_btn(9, 36, "A", g_state.buttons[XINPUT_GAMEPAD_A]);

    // ── Center buttons ───────────────────────────────────────────────────────
    draw_btn(12, 22, "Select", g_state.buttons[XINPUT_GAMEPAD_BACK]);
    // Guide button isn't reliably available in standard XInput, we'll label it
    // but keep it unmapped
    draw_btn(12, 32, "Guide", false);
    draw_btn(12, 41, "Start", g_state.buttons[XINPUT_GAMEPAD_START]);

    // Thumbstick clicks
    draw_btn(13, 2, "L3", g_state.buttons[XINPUT_GAMEPAD_LEFT_THUMB]);
    draw_btn(13, 46, "R3", g_state.buttons[XINPUT_GAMEPAD_RIGHT_THUMB]);

    // ── Divider ──────────────────────────────────────────────────────────────
    goto_xy(15, 1);
    std::printf(
        C_DIM
        "─────────────────────────────────────────────────────────────────────────────\n" RESET);

    // ── Event log ────────────────────────────────────────────────────────────
    goto_xy(16, 2);
    std::printf(C_LABEL BOLD "Recent events" RESET);
    for (int i = 0; i < 5; i++) {
        goto_xy(17 + i, 2);
        if (i < (int)g_state.log.size())
            std::printf(
                C_LABEL "%-76s" RESET,
                g_state.log[g_state.log.size() - 1 - i].c_str());
        else
            std::printf("%-76s", "");
    }

    // ── Footer ───────────────────────────────────────────────────────────────
    goto_xy(23, 1);
    std::printf(C_DIM " Ctrl+C to quit" RESET);

    std::fflush(stdout);
}

// ─────────────────────────────────────────────────────────────────────────────
// XInput Polling Loop
// ─────────────────────────────────────────────────────────────────────────────
void read_xinput() {
    // Look for the first connected controller (User Index 0 to 3)
    int controllerIndex = -1;
    XINPUT_STATE state;

    for (DWORD i = 0; i < 4; i++) {
        std::memset(&state, 0, sizeof(XINPUT_STATE));
        if (XInputGetState(i, &state) == ERROR_SUCCESS) {
            controllerIndex = i;
            g_state.name = "Connected (XInput Controller slot " +
                           std::to_string(i + 1) + ")";
            break;
        }
    }

    if (controllerIndex == -1) {
        std::cout << "No Xbox Controller found. Please plug one in and retry.\n";
        return;
    }

    clear_screen();
    hide_cursor();

    // Track previous button states to log presses/releases properly
    WORD tracked_buttons[] = {
        XINPUT_GAMEPAD_A,
        XINPUT_GAMEPAD_B,
        XINPUT_GAMEPAD_X,
        XINPUT_GAMEPAD_Y,
        XINPUT_GAMEPAD_LEFT_SHOULDER,
        XINPUT_GAMEPAD_RIGHT_SHOULDER,
        XINPUT_GAMEPAD_BACK,
        XINPUT_GAMEPAD_START,
        XINPUT_GAMEPAD_LEFT_THUMB,
        XINPUT_GAMEPAD_RIGHT_THUMB,
        XINPUT_GAMEPAD_DPAD_UP,
        XINPUT_GAMEPAD_DPAD_DOWN,
        XINPUT_GAMEPAD_DPAD_LEFT,
        XINPUT_GAMEPAD_DPAD_RIGHT};

    std::map<WORD, bool> prev_button_states;

    while (g_running) {
        std::memset(&state, 0, sizeof(XINPUT_STATE));
        if (XInputGetState(controllerIndex, &state) != ERROR_SUCCESS) {
            g_state.name = "Controller disconnected.";
            render();
            Sleep(500);
            continue;
        }

        // Update sticks & triggers
        g_state.lx = state.Gamepad.sThumbLX;
        g_state.ly = state.Gamepad.sThumbLY;
        g_state.rx = state.Gamepad.sThumbRX;
        g_state.ry = state.Gamepad.sThumbRY;
        g_state.lt = state.Gamepad.bLeftTrigger;
        g_state.rt = state.Gamepad.bRightTrigger;

        // D-Pad mapper
        g_state.dpx = 0;
        g_state.dpy = 0;
        if (state.Gamepad.wButtons & XINPUT_GAMEPAD_DPAD_LEFT)
            g_state.dpx = -1;
        if (state.Gamepad.wButtons & XINPUT_GAMEPAD_DPAD_RIGHT)
            g_state.dpx = 1;
        if (state.Gamepad.wButtons & XINPUT_GAMEPAD_DPAD_UP)
            g_state.dpy = 1;
        if (state.Gamepad.wButtons & XINPUT_GAMEPAD_DPAD_DOWN)
            g_state.dpy = -1;

        // Check buttons
        for (WORD btn : tracked_buttons) {
            bool is_pressed = (state.Gamepad.wButtons & btn) != 0;
            g_state.buttons[btn] = is_pressed;

            if (is_pressed != prev_button_states[btn]) {
                char log_buf[128];
                std::snprintf(
                    log_buf,
                    sizeof(log_buf),
                    "Button %-8s %s",
                    btn_name(btn).c_str(),
                    is_pressed ? "▼ pressed" : "▲ released");
                g_state.push_log(log_buf);
                prev_button_states[btn] = is_pressed;
            }
        }

        // Handle Trigger logs simply
        static int prev_lt = 0, prev_rt = 0;
        if (std::abs(g_state.lt - prev_lt) > 5) {
            char log_buf[64];
            std::snprintf(
                log_buf,
                sizeof(log_buf),
                "L2 trigger  %d%%",
                g_state.lt * 100 / 255);
            g_state.push_log(log_buf);
            prev_lt = g_state.lt;
        }
        if (std::abs(g_state.rt - prev_rt) > 5) {
            char log_buf[64];
            std::snprintf(
                log_buf,
                sizeof(log_buf),
                "R2 trigger  %d%%",
                g_state.rt * 100 / 255);
            g_state.push_log(log_buf);
            prev_rt = g_state.rt;
        }

        render();
        Sleep(30); // ~30 FPS UI Refresh
    }
}

// ─────────────────────────────────────────────────────────────────────────────
// main
// ─────────────────────────────────────────────────────────────────────────────
int main(int argc, char* argv[]) {
    // Set console control handler for Ctrl+C
    SetConsoleCtrlHandler(console_ctrl_handler, TRUE);

    // Enable Virtual Terminal Processing (ANSI support in Windows Console)
    HANDLE hOut = GetStdHandle(STD_OUTPUT_HANDLE);
    if (hOut != INVALID_HANDLE_VALUE) {
        DWORD dwMode = 0;
        GetConsoleMode(hOut, &dwMode);
        dwMode |= ENABLE_VIRTUAL_TERMINAL_PROCESSING;
        SetConsoleMode(hOut, dwMode);
    }

    read_xinput();

    // Cleanup
    show_cursor();
    clear_screen();
    std::cout << "Bye!\n";
    return 0;
}
